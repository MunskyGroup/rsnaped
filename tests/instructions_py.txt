Save env in file
conda env export > rsnaped.yml
Create env with file
conda env create -f rsnaped.yml


conda create -n rsnaped python=3.8 
conda activate rsnaped
conda deactivate



# create an env

python -m venv env_rsnaped
source env_rsnaped/bin/activate

# to deactivate the env use
deactivate

# create distrubution
python -c 'import setuptools'
python setup.py sdist

# check dist file
tar --list -f dist/rsnaped-0.0.1.tar.gz

# install on local computer
pip install -e .

# to push into pypi
twine upload dist/*
